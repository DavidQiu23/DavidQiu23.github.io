## 繼承 Application 以管理全域 App 狀態
透過繼承 [Application()](https://developer.android.com/reference/android/app/Application) 類別，可以集中管理 App 的生命週期與全域資料。

### 全域變數與 SharedPreferences
```kotlin 
lateinit var settings: SharedPreferences
```

### 紀錄目前作用中的 Activity
```kotlin 
lateinit var currentActivity: Activity // 用於快速取得目前顯示的 Activity 物件
```

### 全域變數、常數與擴充函式 (Extensions)
```kotlin 
// 常數定義
const val DATA = "DATA"

// 全域變數
var func = arrayListOf<String>()

// 日期格式化工具
fun setDateFormat(year: Int, monthOfYear: Int, dayOfMonth: Int): String {
    return (year.toString() + "/"
            + timeFix(monthOfYear + 1) + "/"
            + timeFix(dayOfMonth))
}

// AES 加密擴充函式
fun String.encrypt(): String {
    val cipher = Cipher.getInstance("AES/ECB/PKCS5Padding")
    val raw = KEY.toByteArray(charset("utf-8"))
    val skeySpec = SecretKeySpec(raw, "AES")
    cipher.init(Cipher.ENCRYPT_MODE, skeySpec)
    val cipherText = cipher.doFinal(this.toByteArray())
    return Base64.getEncoder().encodeToString(cipherText)
}
```

### 配置 OkHttp3 客戶端
建議在 App 內全域共用單一 `OkHttpClient` 實例：
```kotlin
val client = OkHttpClient.Builder()
    .addInterceptor(HttpLoggingInterceptor().setLevel(HttpLoggingInterceptor.Level.BODY))
    .eventListener(RequestEventListener())
    .build()
```
- **HttpLoggingInterceptor**: 用於開發時紀錄請求與回應的詳細資訊。
- **EventListener**: 可針對請求開始、成功、失敗等事件進行全域監控或處置。

## 工具類別 (Utility Classes)

### DialogUtil.kt：通用對話框工具
```kotlin
companion object {
    private lateinit var progressDialog: AlertDialog
    private var progressCount = 0

    // 顯示訊息對話框
    fun showDialog(context: Context?, msg: String, title: String = "提示", isBack: Boolean = false, isCancel: Boolean = false, func: () -> Unit = {}) {
        val builder = AlertDialog.Builder(context!!)
        builder.setMessage(msg).setTitle(title)
        
        builder.setPositiveButton("OK") { dialog, _ ->
            func()
            dialog.dismiss()
            if (isBack) (context as Activity).finish()
        }
        
        if (isCancel) {
            builder.setNegativeButton("Cancel") { dialog, _ -> dialog.dismiss() }
        }
        
        (context as Activity).runOnUiThread {
            builder.setCancelable(false)
            builder.create().show()
        }
    }

    // 顯示登出確認對話框
    fun showLogoutDialog(context: Context) {
        AlertDialog.Builder(context)
            .setTitle("Logout")
            .setMessage("Are you sure you want to logout?")
            .setPositiveButton("Yes") { _, _ -> (context as Activity).finishAffinity() }
            .setNegativeButton("No", null)
            .show()
    }

    // 顯示讀取進度對話框
    fun showProgressDialog(context: Context, title: String = "處理中，請稍候...", msg: String = "") {
        if (progressCount == 0) {
            val builder = AlertDialog.Builder(context)
            builder.setTitle(title)
            val v = LayoutInflater.from(context).inflate(R.layout.progress, null)
            v.findViewById<TextView>(R.id.txt_desc).text = msg
            builder.setView(v)

            (context as Activity).runOnUiThread {
                if (!this::progressDialog.isInitialized || !progressDialog.isShowing) {
                    progressDialog = builder.create()
                    progressDialog.setCanceledOnTouchOutside(false)
                }
                progressDialog.show()
            }
        }
        progressCount++
    }

    // 關閉讀取進度對話框
    fun dismissProgressDialog(context: Context) {
        progressCount--
        if (progressCount <= 0) {
            progressCount = 0
            (context as Activity).runOnUiThread {
                if (this::progressDialog.isInitialized && progressDialog.isShowing) {
                    progressDialog.dismiss()
                }
            }
        }
    }
}
```

## 異常捕獲與錯誤處理
在發生閃退時，記錄詳細錯誤資訊並將 App 重新引導回錯誤提示頁面，避免 ANR 或全域當機。

```kotlin
val oldHandler = Thread.getDefaultUncaughtExceptionHandler()

Thread.setDefaultUncaughtExceptionHandler { paramThread, paramThrowable ->
    // 1. 寫入錯誤 Log 到本地檔案
    val file = File(filesDir, "error.txt")
    file.writeText("${LocalDateTime.now()} | User: $userId\n${paramThrowable.stackTraceToString()}\n")

    // 2. 判斷錯誤類型並導向錯誤頁面
    val errorMsg = if (paramThrowable.stackTraceToString().contains("\"status\":403")) {
        "認證過期，請重新登入！"
    } else {
        "發生未知錯誤，請聯繫管理員。"
    }

    val intent = Intent(applicationContext, ErrorActivity::class.java).apply {
        putExtra("errorMsg", errorMsg)
        addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP or Intent.FLAG_ACTIVITY_NEW_TASK)
    }
    startActivity(intent)

    // 3. 交由原處理程序處理或強制退出
    if (oldHandler != null) {
        oldHandler.uncaughtException(paramThread, paramThrowable)
    } else {
        exitProcess(1)
    }
}
```
